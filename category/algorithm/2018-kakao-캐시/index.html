import collections
def solution(cacheSize, cities):
  answer = 0
  caches = collections.deque()

  for city in cities:
      city = city.lower()
      # cache hit
      if city in caches:
          caches.remove(city)
          caches.append(city)
          answer += 1
      # cache miss
      else:
          if cacheSize == 0:
              return 5 * len(cities)
          # ìµœê·¼ ì‚¬ìš©í•˜ì§€ ì•Šì€ ìºì‹œ ì—…ë°ì´íŠ¸
          if len(caches) == cacheSize:
              caches.popleft()
          caches.append(city)
          answer += 5
  return answer
